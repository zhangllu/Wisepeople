### app/wise-persons/[slug]/page.tsx

```typescript
//智者介绍页
import { notFound } from 'next/navigation';
import { wisePeople } from '@/lib/metadata';

export async function generateStaticParams() {
  return wisePeople.map((person) => ({
    slug: person.slug,
  }));
}

export default function WisePersonPage({ params }: { params: { slug: string } }) {
  const person = wisePeople.find(p => p.slug === params.slug);

  if (!person) {
    notFound();
  }

  // 从Markdown文件读取内容
  const content = await getMarkdownContent(person.slug, '01-智者介绍.md');

  return (
    <div>
      <h1>{person.name}</h1>
      <h2>{person.english_name}｜{person.years}</h2>

      {/* 页面内导航 */}
      <nav>
        <a href={`/wise-persons/${person.slug}`}>介绍</a>
        <a href={`/wise-persons/${person.slug}/sources`}>原始资料</a>
        <a href={`/wise-persons/${person.slug}/cognitive-style`}>认知方式</a>
      </nav>

      {/* Markdown内容渲染 */}
      <article>{content}</article>
    </div>
  );
}
```

## 2. Markdown内容读取工具

```typescript
// lib/markdown.ts
import fs from 'fs';
import path from 'path';

const WISE_PEOPLE_BASE_PATH = '/Users/zhanglu/Desktop/Wisepeople/智者资料库';

export async function getMarkdownContent(slug: string, filename: string) {
  const filePath = path.join(WISE_PEOPLE_BASE_PATH, slug, filename);

  try {
    const content = fs.readFileSync(filePath, 'utf-8');
    // 使用你选择的Markdown渲染器
    return renderMarkdown(content);
  } catch (error) {
    return null;
  }
}

function renderMarkdown(content: string) {
  // 可以选择：
  // - next-mdx-remote
  // - react-markdown
  // - marked
  // 或者自己实现
  return content;
}
```

## 4. 问题分类页面

```typescript
// app/wise-persons/question/[id]/page.tsx
export default function QuestionPage({ params }: { params: { id: string } }) {
  const question = questions.find(q => q.id === params.id);
  const people = getWisePeopleByQuestion(params.id);

  return (
    <div>
      <h1>{question.zh_title}</h1>
      <p>{question.description}</p>

      {/* 如果有子分类 */}
      {question.subcategories && (
        <div>
          {question.subcategories.map(sub => (
            <section key={sub.id}>
              <h2>{sub.zh_title}</h2>
              <div className="wise-people-grid">
                {people
                  .filter(p => p.subcategory === sub.id)
                  .map(person => (
                    <WisePersonCard key={person.slug} person={person} />
                  ))}
              </div>
            </section>
          ))}
        </div>
      )}

      {/* 没有子分类，直接显示 */}
      {!question.subcategories && (
        <div className="wise-people-grid">
          {people.map(person => (
            <WisePersonCard key={person.slug} person={person} />
          ))}
        </div>
      )}
    </div>
  );
}

function WisePersonCard({ person }: { person: WisePerson }) {
  return (
    <Link href={`/wise-persons/${person.slug}`}>
      <div className="card">
        <h3>{person.name}</h3>
        <p>{person.english_name}｜{person.years}</p>
        <div className="topics">
          {person.topics.slice(0, 3).map(topic => (
            <span key={topic} className="topic-tag">{topic}</span>
          ))}
        </div>
        <div className="page-links">
          <a href={`/wise-persons/${person.slug}`}>介绍</a>
          {person.has_sources && (
            <a href={`/wise-persons/${person.slug}/sources`}>原始资料</a>
          )}
          {person.has_cognitive_style && (
            <a href={`/wise-persons/${person.slug}/cognitive-style`}>认知方式</a>
          )}
        </div>
      </div>
    </Link>
  );
}
```

## 5. 页面内导航组件

```typescript
// components/WisePersonNav.tsx
interface Props {
  slug: string;
  activeTab: 'introduction' | 'sources' | 'cognitive-style';
}

export function WisePersonNav({ slug, activeTab }: Props) {
  const tabs = [
    { id: 'introduction', label: '介绍', href: `/wise-persons/${slug}` },
    { id: 'sources', label: '原始资料', href: `/wise-persons/${slug}/sources` },
    { id: 'cognitive-style', label: '认知方式', href: `/wise-persons/${slug}/cognitive-style` },
  ];

  return (
    <nav className="wise-person-nav">
      {tabs.map(tab => (
        <a
          key={tab.id}
          href={tab.href}
          className={tab.id === activeTab ? 'active' : ''}
        >
          {tab.label}
        </a>
      ))}
    </nav>
  );
}
```

## 6. 面包屑导航

```typescript
// components/Breadcrumb.tsx
interface Props {
  questionId?: string;
  slug: string;
  activeTab: string;
}

export function Breadcrumb({ questionId, slug, activeTab }: Props) {
  const person = wisePeople.find(p => p.slug === slug);
  const question = questionId ? questions.find(q => q.id === questionId) : null;

  const tabNames = {
    introduction: '介绍',
    sources: '原始资料',
    'cognitive-style': '认知方式'
  };

  return (
    <nav className="breadcrumb">
      <a href="/">首页</a>
      {' > '}
      <a href="/wise-persons">智者库</a>
      {question && (
        <>
          {' > '}
          <a href={`/wise-persons/question/${questionId}`}>
            {question.zh_title}
          </a>
        </>
      )}
      {' > '}
      <span>{person.name}</span>
      {' > '}
      <span>{tabNames[activeTab]}</span>
    </nav>
  );
}
```

## 8. 搜索和筛选功能

```typescript
// app/wise-persons/page.tsx
'use client';

import { useState } from 'react';

export default function WisePeopleListPage() {
  const [searchQuery, setSearchQuery] = useState('');
  const [selectedQuestion, setSelectedQuestion] = useState<string | null>(null);

  const filteredPeople = wisePeople.filter(person => {
    const matchesSearch =
      person.name.includes(searchQuery) ||
      person.english_name.toLowerCase().includes(searchQuery.toLowerCase()) ||
      person.topics.some(topic => topic.includes(searchQuery));

    const matchesQuestion =
      !selectedQuestion || person.questions.includes(selectedQuestion);

    return matchesSearch && matchesQuestion;
  });

  return (
    <div>
      {/* 搜索框 */}
      <input
        type="text"
        placeholder="搜索智者、主题..."
        value={searchQuery}
        onChange={(e) => setSearchQuery(e.target.value)}
        className="search-input"
      />

      {/* 问题筛选 */}
      <div className="question-filters">
        <button onClick={() => setSelectedQuestion(null)}>
          全部
        </button>
        {questions.map(q => (
          <button
            key={q.id}
            onClick={() => setSelectedQuestion(q.id)}
            className={selectedQuestion === q.id ? 'active' : ''}
          >
            {q.zh_title}
          </button>
        ))}
      </div>

      {/* 结果列表 */}
      <div className="wise-people-grid">
        {filteredPeople.map(person => (
          <WisePersonCard key={person.slug} person={person} />
        ))}
      </div>
    </div>
  );
}
```

## 9. Sitemap生成

```typescript
// app/sitemap.ts
import { MetadataRoute } from 'next';
import { wisePeople, questions } from '@/lib/metadata';

export default function sitemap(): MetadataRoute.Sitemap {
  const baseUrl = 'https://wisepeople.vercel.app';

  const wisePersonUrls = wisePeople.flatMap(person => [
    {
      url: `${baseUrl}/wise-persons/${person.slug}`,
      lastModified: new Date(),
    },
    ...(person.has_sources ? [{
      url: `${baseUrl}/wise-persons/${person.slug}/sources`,
      lastModified: new Date(),
    }] : []),
    ...(person.has_cognitive_style ? [{
      url: `${baseUrl}/wise-persons/${person.slug}/cognitive-style`,
      lastModified: new Date(),
    }] : []),
  ]);

  const questionUrls = questions.map(question => ({
    url: `${baseUrl}/wise-persons/question/${question.id}`,
    lastModified: new Date(),
  }));

  return [
    { url: baseUrl, lastModified: new Date() },
    { url: `${baseUrl}/wise-persons`, lastModified: new Date() },
    ...questionUrls,
    ...wisePersonUrls,
  ];
}
```

## 总结

这个网页设计方案的核心特点：

1. **清晰的路由结构**：每个智者有3个子页面（介绍、原始资料、认知方式）
2. **灵活的分类**：按问题（Q01-Q05）组织智者
3. **页面内导航**：Tab或侧边栏切换，用户友好
4. **面包屑导航**：用户知道当前位置
5. **搜索和筛选**：方便用户找到感兴趣的智者
6. **SEO友好**：每个页面有独立URL，利于搜索引擎收录
